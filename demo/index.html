import Stats from 'stats-gl'

const stats = new Stats();
const renderer = new THREE.WebGLRenderer();
stats.init( renderer.domElement );

scene.onBeforeRender = function () {
    stats.begin();
};

scene.onAfterRender = function () {
    stats.end();
};
